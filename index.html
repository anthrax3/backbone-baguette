Backbone Baguette

Backbone on steroids.

Write dramatically less code, enjoy your app more. Baguette is only 1.4Kb.

Model Views

Auto-binding and re-rendering, done under the hood.

// Create a simple model
var car = new Backbone.Model({brand:"Peugeot", name:"405"});

// Create a new Model View with a mustache-type template.
var ModelView = Backbone.Baguette.ModelView.extend({
  tpl:"My car is a {{brand}} {{name}}"
});

// Instantiate our view
var myView = new ModelView({model:car});

// "My car is a Peugeot 405" inserted to body
myView.render().$el.appendTo('body');

// View refreshed "My car is a Peugeot 508"
car.set({name:'508'});

Collection Views

Auto-bindind and re-rendering when collection changes. "Memory leaks free" view management.

// Create some dope cars
var peugeot = new Backbone.Model({brand:"Peugeot", name:"508"});
var renault = new Backbone.Model({brand:"Renault", name:"11"});

// Let's build our garage
var myGarage = new Backbone.Collection([peugeot,renault]);

// Create a view for cars
var CarView = Backbone.Baguette.ModelView.extend({
  tpl:"My car is a {{brand}} {{name}}"
});

// Create our garage view
var MyGarageView = Backbone.Baguette.CollectionView.extend({
  modelView:CarView,
  loader:true
});

// Instantiate the garage view
var myGarageView = new GarageView({collection:myGarage});
myGarageView.render().$el.appendTo('body');
// ^ My two CarViews are appended to the DOM
// ^ Wrapped by the GarageView element (div by default)

// Let's add a new model
myGarage.add(new Backbone.Model({brand:"CitroÃ«n", name:"C3"}));
// ^ View refreshed, the new CarView is added!

// Let's edit a model of the collection
peugeot.set({name:"RCZ"});
// ^ Sub-model view refreshed, without repainting the whole collection.

// Fetch data from the server
myGarage.fetch();
// ^ Loader is popping up untill data comes! View refreshed after fetch.

// Let's burn this garage, mafia style.
myGarageView.remove();
// ^ All inner views (model views) are properly removed to avoid memory leaks

Composite Views

Handle complex data and view structures. Without the hassle.

// Create a basic model
var complexModel = new Backbone.Model({
  simpleString:"hello"
});
// Add some complexity to one of its attribute : let's have a nested model.
complexModel.set('subModel', new Backbone.Model({subAttribute:"I'm a sub model!"}));

// Create our view for our sub model
var SubModelView = Backbone.Baguette.ModelView.extend({
  tpl:"This is my attribute value : {{subAttribute}}"
});

// Create the main view
var ComplexModelView = Backbone.Baguette.CompositeView.extend({
  tagName:"p",
  tpl:'<div class="header">Welcome {{simpleString}}</div><div class="content"></div>',
  nestedViews:{
    // CSS-style selectors
    '.content':{
      view:SubModelView,
      modelAttribute:'subModel'
    }
  }
});
var cmv = new ComplexModelView({model:complexModel});
cmv.render().$el.appendTo('body');
// Inserted to body:
//<p>
//<div class="header">Welcome hello</div>
//<div class="content">This is my attribute value : I'm a sub model!</div>
//</p>
